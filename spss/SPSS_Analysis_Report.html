# ─── SPSS: Analyze > Descriptive Statistics > Frequencies ───

def freq_table(series, label):
    freq = series.value_counts().reset_index()
    freq.columns = [label, 'Frequency']
    freq['Percent (%)'] = (freq['Frequency'] / len(series) * 100).round(2)
    freq['Cumulative (%)'] = freq['Percent (%)'].cumsum().round(2)
    return freq

# Three frequency tables
for col in ['Segment', 'Region', 'Category']:

    print(f"FREQUENCY TABLE — {col}")

# ─── SPSS: Analyze > Descriptive Statistics > Explore > Normality ───
# SPSS uses Shapiro-Wilk (small n) and K-S Test (large n). We use both.

print("=" * 65)
print("NORMALITY TESTS — Shapiro-Wilk & Kolmogorov-Smirnov")
print("=" * 65)

test_cols = ['Sales', 'Profit', 'Discount', 'Profit Margin %', 'Ship Days']

results = []
for col in test_cols:
    data = df[col].dropna()
    sample = data.sample(500, random_state=42)  # Shapiro-Wilk: max 5000 recommended

    sw_stat, sw_p = stats.shapiro(sample)
    ks_stat, ks_p = stats.kstest(data, 'norm', args=(data.mean(), data.std()))

    results.append({
        'Variable': col,
        'SW Statistic': round(sw_stat, 4),
        'SW p-value': round(sw_p, 4),
        'SW Decision': 'NOT Normal' if sw_p < 0.05 else 'Normal',
        'KS Statistic': round(ks_stat, 4),
        'KS p-value': round(ks_p, 4),
        'KS Decision': 'NOT Normal' if ks_p < 0.05 else 'Normal',
    })

norm_df = pd.DataFrame(results)
print(norm_df.to_string(index=False))

print("\n📌 Interpretation:")
print("   p < 0.05 → Reject H₀ → Variable is NOT normally distributed")
print("   For non-normal variables, prefer Spearman over Pearson, Mann-Whitney over T-Test")

# ─── SPSS: Analyze > Compare Means > Independent Samples T-Test ───

profitable = df[df['Is Profitable'] == 1]['Profit Margin %'].dropna()
not_profitable = df[df['Is Profitable'] == 0]['Profit Margin %'].dropna()

print("=" * 65)
print("INDEPENDENT T-TEST — Profit Margin % by Profitability Status")
print("=" * 65)
print(f"Profitable Orders     : n={len(profitable):,}, mean={profitable.mean():.2f}%, std={profitable.std():.2f}")
print(f"Non-Profitable Orders : n={len(not_profitable):,}, mean={not_profitable.mean():.2f}%, std={not_profitable.std():.2f}")

# Step 1 — Levene's Test
lev_stat, lev_p = stats.levene(profitable, not_profitable)
equal_var = lev_p >= 0.05
print(f"\nLevene's Test for Equal Variances:")
print(f"  F = {lev_stat:.4f}, p = {lev_p:.4f} → {'Equal variances assumed' if equal_var else 'Equal variances NOT assumed'}")

# Step 2 — T-Test (version-safe: no hardcoded keyword or column names)
ttest = pg.ttest(profitable, not_profitable, correction=not equal_var)
print(f"\nT-Test Results:")
print(ttest.to_string())

print(f"\n📌 Interpretation:")
print(f"   p < 0.05 → Significant difference in Profit Margin % between groups")
print(f"   Cohen's d → Small=0.2, Medium=0.5, Large=0.8")
